SELECT IL.NAME, I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUESLISTS IL, ISSUES I, USERS U
WHERE I.ISSUESLIST_ID = IL.ID
AND I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_ASSIGNED_TO
FROM USERS U, ISSUES I
WHERE I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) >= 1;
